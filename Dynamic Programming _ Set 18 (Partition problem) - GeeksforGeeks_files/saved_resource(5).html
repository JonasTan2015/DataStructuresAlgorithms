                                                                                    Rather than doing all this a very simple solution is just sort the array and subtract last element from second last, take absolute value of that and subtract from third last then take absolute of that and do until first element.If the final answer is 0(zero),then it can be divided into two subsets such that the sum of elements in both subsets is same.
                                                                                    CODE:

                                                                                    def findpartition(arr):
                                                                                    val = arr[n-1]
                                                                                    for i in range(0,n-1):
                                                                                    val = abs(val-arr[n-2-i]) #absolute value of the difference
                                                                                    if val==0:
                                                                                    print "Array can be divided into two subsets of equal sum "
                                                                                    else:
                                                                                    print "Array can not be divided into two subsets of equal sum"

                                                                                    ##below is quick sort algorithm
                                                                                    def partition(arr,low,high):
                                                                                    i = ( low-1 ) # index of smaller element
                                                                                    pivot = arr[high] # pivot

                                                                                    for j in range(low , high):

                                                                                    # If current element is smaller than or
                                                                                    # equal to pivot
                                                                                    if arr[j] <= pivot:

                                                                                    # increment index of smaller element
                                                                                    i = i+1
                                                                                    arr[i],arr[j] = arr[j],arr[i]

                                                                                    arr[i+1],arr[high] = arr[high],arr[i+1]
                                                                                    return ( i+1 )

                                                                                    def quickSort(arr,low,high):
                                                                                    if low < high:

                                                                                    # pi is partitioning index, arr[p] is now
                                                                                    # at right place
                                                                                    pi = partition(arr,low,high)

                                                                                    # Separately sort elements before
                                                                                    # partition and after partition
                                                                                    quickSort(arr, low, pi-1)
                                                                                    quickSort(arr, pi+1, high)

                                                                                    arr = [13,4,12,3,1,17,8]
                                                                                    n=len(arr)
                                                                                    quickSort(arr,0,n-1)

                                                                                            bool equalSumSubarray(int [] a)

                                                                                            {

                                                                                            int sum = 0;

                                                                                            for (int i = 0; i < a.Length; i++) {

                                                                                            sum += a [i];

                                                                                            }

                                                                                            if (sum % 2 != 0)

                                                                                            return false;

                                                                                            int half = sum / 2;

                                                                                            int index = 0;

                                                                                            return equalSumSubUtil (a, half, index);

                                                                                            } //Main Function

                                                                                            bool equalSumSubUtil(int []a, int halfSum, int i)

                                                                                            {

                                                                                            if (halfSum == 0)

                                                                                            return true;

                                                                                            if (i == a.Length)

                                                                                            return false;

                                                                                            if (a [i] < halfSum)

                                                                                            halfSum = halfSum - a [i];

                                                                                            else if (a [i] == halfSum)

                                                                                            return true;

                                                                                            return equalSumSubUtil (a, halfSum, i++);

                                                                                            } //Utility Function
