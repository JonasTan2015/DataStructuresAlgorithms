                                                                                                                                                              def longest_increasing_sub_sequence(a, offset=0, prev=-1, c=0):

                                                                                                                                                              if offset >= len(a):

                                                                                                                                                              return c

                                                                                                                                                              if prev < 0:

                                                                                                                                                              take_it = 1

                                                                                                                                                              else:

                                                                                                                                                              if a[offset] > a[prev]:

                                                                                                                                                              take_it += 1
